namespace base {
namespace backport {

template <typename T, T... Indices>
struct integer_sequence {
  using value_type = T;
  static constexpr std::size_t size() noexcept;
};

template <typename T, T N>
using make_integer_sequence = integer_sequence<T, /* 0, 1, 2, ..., (N - 1)*/ >;

template <std::size_t... Indices>
using index_sequence = integer_sequence<std::size_t, Indices...>;

template <std::size_t N>
using make_index_sequence = make_integer_sequence<std::size_t, N>;

}
}

FROZEN. These are backports of std::integer_sequence and friends from C++14. They provide compile-time integer sequences, which are useful for unpacking items from a std::tuple.
