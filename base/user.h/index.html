namespace base {

struct User {
  int32_t uid;
  int32_t gid;
  std::string name;
  std::string gecos;
  std::string homedir;
  std::string shell;

  User(int32_t uid = -1,
       int32_t gid = -1,
       std::string name = std::string(),
       std::string gecos = std::string(),
       std::string homedir = std::string(),
       std::string shell = std::string()) noexcept;
  User(const User& other);
  User(User&& other) noexcept;
  User& operator=(const User& other);
  User& operator=(User&& other) noexcept;

  void reset() noexcept;
  void swap(User& other) noexcept;

  void append_to(std::string* out) const;
  std::size_t length_hint() const noexcept;
  std::string as_string() const;
};

void swap(User& a, User& b) noexcept;
bool operator==(const User& a, const User& b) noexcept;
bool operator!=(const User& a, const User& b) noexcept;

}
